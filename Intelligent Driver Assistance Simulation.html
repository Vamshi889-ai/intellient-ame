🔹 main.py: Driving Simulator Logic

python
import pygame
import random
import time

# Init
pygame.init()
screen = pygame.display.set_mode((800, 600))
clock = pygame.time.Clock()

# Vehicle Properties
vehicle = pygame.Rect(100, 300, 50, 30)
speed = 0
ideal_speed = 50
score = 100

font = pygame.font.SysFont(None, 30)

def show_text(text, x, y):
    img = font.render(text, True, (255, 255, 255))
    screen.blit(img, (x, y))

# Game Loop
running = True
while running:
    screen.fill((0, 0, 50))
    keys = pygame.key.get_pressed()

    # Simulated sensor input
    if keys[pygame.K_UP]:
        speed += 1
    if keys[pygame.K_DOWN]:
        speed -= 1
    if keys[pygame.K_SPACE]:
        speed = max(0, speed - 5)  # Braking

    # Draw vehicle
    pygame.draw.rect(screen, (0, 255, 0), vehicle)
    vehicle.x += speed // 5

    # Show data
    show_text(f"Speed: {speed} km/h", 10, 10)
    show_text(f"Ideal Speed: {ideal_speed} km/h", 10, 40)

    # Overtake logic (basic)
    if speed > ideal_speed + 10:
        score -= 1
        show_text("Unsafe Overtaking!", 300, 50)

    show_text(f"Performance Score: {score}", 10, 70)

    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    pygame.display.flip()
    clock.tick(30)

pygame.quit()

🔹 replay.py: Journey Replay Visualizer (Matplotlib)

python
import matplotlib.pyplot as plt

# Simulated journey data
times = list(range(0, 100))
speeds = [random.randint(40, 80) for _ in times]
ideal = [60] * 100

plt.plot(times, speeds, label='Driver Speed')
plt.plot(times, ideal, '--', label='Ideal Speed')
plt.title("Journey Speed Analysis")
plt.xlabel("Time (s)")
plt.ylabel("Speed (km/h)")
plt.legend()
plt.grid(True)
plt.savefig("replay.png")
plt.show()

🔹 map_sim.py: Google Maps Integration (Static)

python
import folium

# Simulated route (lat/lng)
start = [37.7749, -122.4194]
end = [37.8044, -122.2711]

m = folium.Map(location=start, zoom_start=13)
folium.Marker(start, tooltip="Start").add_to(m)
folium.Marker(end, tooltip="End").add_to(m)
folium.PolyLine([start, end], color="blue", weight=2.5, opacity=1).add_to(m)

m.save("drive_path.html")
